    **1. Create `scripts/seed-admin.ts`:**

    ```typescript
    /**
     * Seed an admin user for Pick a Partner.
     * Usage: npm run seed:admin
     * Requires SUPABASE_SERVICE_ROLE_KEY in .env.local
     */
    import { createClient } from "@supabase/supabase-js";
    import "dotenv/config"; // Load .env.local

    const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL;
    const serviceRoleKey = process.env.SUPABASE_SERVICE_ROLE_KEY;

    if (!supabaseUrl || !serviceRoleKey) {
      console.error("Missing NEXT_PUBLIC_SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY");
      console.error("Add them to .env.local");
      process.exit(1);
    }

    const supabase = createClient(supabaseUrl, serviceRoleKey, {
      auth: { autoRefreshToken: false, persistSession: false },
    });

    const ADMIN_EMAIL = "admin@pickapartner.app";
    const ADMIN_PASSWORD = "admin123456";

    async function seedAdmin() {
      console.log("Seeding admin user...");

      // Try to create auth user
      const { data: authData, error: authError } = await supabase.auth.admin.createUser({
        email: ADMIN_EMAIL,
        password: ADMIN_PASSWORD,
        email_confirm: true,
      });

      let userId: string;

      if (authError) {
        if (authError.message.includes("already been registered")) {
          console.log("Auth user already exists, finding...");
          const { data: { users } } = await supabase.auth.admin.listUsers();
          const existing = users?.find((u) => u.email === ADMIN_EMAIL);
          if (!existing) {
            console.error("Could not find existing admin auth user");
            process.exit(1);
          }
          userId = existing.id;
        } else {
          console.error("Failed to create auth user:", authError.message);
          process.exit(1);
        }
      } else {
        userId = authData.user.id;
      }

      // Upsert player profile with admin role
      const { error: playerError } = await supabase.from("players").upsert(
        { user_id: userId, display_name: "Admin", role: "admin" },
        { onConflict: "user_id" }
      );

      if (playerError) {
        console.error("Failed to upsert player:", playerError.message);
        process.exit(1);
      }

      console.log("Admin user ready!");
      console.log(`  Email: ${ADMIN_EMAIL}`);
      console.log(`  Password: ${ADMIN_PASSWORD}`);
    }

    seedAdmin().catch((err) => {
      console.error("Seed failed:", err);
      process.exit(1);
    });
    ```

    **2. Install tsx and dotenv as devDependencies if not present:**
    ```bash
    npm install -D tsx dotenv
    ```

    **3. Add script to package.json:**
    ```json
    "seed:admin": "tsx --env-file=.env.local scripts/seed-admin.ts"
    ```
    Note: tsx >= 4.x supports `--env-file`. If using older tsx, use `dotenv/config` import instead
    (which is already in the script as a fallback). Either approach works.
    Actually, Node 20+ supports `--env-file` natively. Use:
    ```json
    "seed:admin": "npx tsx scripts/seed-admin.ts"
    ```
    And rely on the `dotenv/config` import in the script to load `.env.local`. BUT `dotenv` does
    not load `.env.local` by default -- it loads `.env`. So either:
    - Option A: Use `dotenv` with explicit path: `import dotenv from "dotenv"; dotenv.config({ path: ".env.local" });`
    - Option B: Use Next.js env loading: won't work outside Next.js context
    - Option C: Use `--env-file=.env.local` flag with Node 20+: `"seed:admin": "node --env-file=.env.local node_modules/.bin/tsx scripts/seed-admin.ts"`

    Best approach: Replace the `import "dotenv/config"` with explicit dotenv config at the top of
    the script:
    ```typescript
    import { config } from "dotenv";
    config({ path: ".env.local" });
    ```
    And install dotenv: `npm install -D dotenv`
    Package.json script: `"seed:admin": "npx tsx scripts/seed-admin.ts"`

    **4. Run the seed script** (if Supabase credentials are available):
    ```bash

    ```
