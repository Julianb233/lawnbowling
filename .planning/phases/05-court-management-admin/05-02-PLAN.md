         const fetchMatches = async () => {
           setLoading(true);
           const supabase = createClient();
           let query = supabase
             .from("matches")
             .select("*, courts(name), match_players(player_id, team, players(display_name, avatar_url))", { count: "exact" })

           if (statusFilter) query = query.eq("status", statusFilter);

           setTotal(count ?? 0);
           setLoading(false);
         };
