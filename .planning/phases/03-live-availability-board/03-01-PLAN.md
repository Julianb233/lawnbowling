1. Create `src/app/api/checkin/route.ts` with a POST handler:
   - Import `createClient` from `@/lib/supabase/server`
   - Get the authenticated user via `supabase.auth.getUser()` -- return 401 JSON `{ error: "Not authenticated" }` if no user
   - Fetch the current player row: `.from("players").select("id, is_available").eq("user_id", user.id).single()`
   - Return 404 if no player found
   - Toggle: set `is_available` to `!player.is_available`
   - If checking IN (new is_available = true): set `checked_in_at` to `new Date().toISOString()`
   - If checking OUT (new is_available = false): set `checked_in_at` to `null`
   - Update: `.update({ is_available, checked_in_at }).eq("id", player.id).select().single()`
   - Return `NextResponse.json({ is_available, checked_in_at })` with status 200
   - Wrap in try/catch, return `{ error: message }` with status 500 on failure

     ```
     setAvailable(data.is_available);
     onToggle?.(data.is_available);
     ```
   - Remove the manual `newState` calculation since the server determines the toggled state
   - Keep ALL existing animation/UI code unchanged (motion, AnimatePresence, shimmer, etc.)

2. **Append to `src/lib/db/schema.sql`** (after existing content):
   ```sql
   -- ===== Realtime & RLS for Phase 3: Live Availability Board =====

   -- Enable Realtime publication for players table (required for postgres_changes)
   alter publication supabase_realtime add table players;

   -- Enable Row Level Security
   alter table players enable row level security;

   -- Anyone authenticated can read available players (for the board)
   create policy "read_available_players" on players
     for select using (is_available = true);

   -- Players can read their own row regardless of availability
   create policy "read_own_profile" on players
     for select using (auth.uid() = user_id);

   -- Players can update their own row (check-in/out, profile)
   create policy "update_own_profile" on players
     for update using (auth.uid() = user_id)
     with check (auth.uid() = user_id);

   -- Players can insert their own row (registration)
   create policy "insert_own_profile" on players
     for insert with check (auth.uid() = user_id);
   ```
  </action>
